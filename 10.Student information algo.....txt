import java.sql.*;
import java.util.Scanner;

class StudentJDBC
{
    public static void main(String args[])
    {
        Scanner sc = new Scanner(System.in);

        try
        {
            Class.forName("com.mysql.cj.jdbc.Driver");

            Connection con = DriverManager.getConnection(
                    "jdbc:mysql://localhost:3306/college",
                    "root",
                    "avcoe");

            int choice;

            do
            {
                System.out.println("\n1.Add");
                System.out.println("2.Update");
                System.out.println("3.Delete");
                System.out.println("4.View");
                System.out.println("5.Exit");

                System.out.print("Enter Choice: ");
                choice = sc.nextInt();

                switch(choice)
                {
                    case 1:

                        System.out.print("Enter Roll No: ");
                        int roll = sc.nextInt();
                        sc.nextLine();

                        System.out.print("Enter Name: ");
                        String name = sc.nextLine();

                        PreparedStatement ps1 = con.prepareStatement(
                                "insert into student values(?,?)");

                        ps1.setInt(1, roll);
                        ps1.setString(2, name);

                        con.setAutoCommit(false);

                        ps1.executeUpdate();

                        con.commit();

                        System.out.println("Student Added");
                        break;

                    case 2:

                        System.out.print("Enter Roll No: ");
                        int r1 = sc.nextInt();
                        sc.nextLine();

                        System.out.print("Enter New Name: ");
                        String newname = sc.nextLine();

                        PreparedStatement ps2 = con.prepareStatement(
                                "update student set name=? where rollno=?");

                        ps2.setString(1, newname);
                        ps2.setInt(2, r1);

                        ps2.executeUpdate();

                        System.out.println("Student Updated");
                        break;

                    case 3:

                        System.out.print("Enter Roll No: ");
                        int r2 = sc.nextInt();

                        PreparedStatement ps3 = con.prepareStatement(
                                "delete from student where rollno=?");

                        ps3.setInt(1, r2);

                        ps3.executeUpdate();

                        System.out.println("Student Deleted");
                        break;

                    case 4:

                        PreparedStatement ps4 = con.prepareStatement(
                                "select * from student");

                        ResultSet rs = ps4.executeQuery();

                        while(rs.next())
                        {
                            System.out.println(
                                    rs.getInt(1) + " " +
                                    rs.getString(2));
                        }

                        break;

                    case 5:

                        System.out.println("Program End");
                        break;

                    default:

                        System.out.println("Invalid Choice");
                }

            } while(choice != 5);

            con.close();
        }

        catch(Exception e)
        {
            System.out.println(e);
        }
    }
}

JDBC CRUD Operations Using Java and MySQL

Title

Develop a Java application that connects to a MySQL database to manage student information using JDBC and performs CRUD operations (Create, Read, Update, Delete).


---

1. Aim

To develop a Java application using JDBC that connects with a MySQL database and performs CRUD operations on student records using prepared statements and transaction management.


---

2. Objective

To understand database connectivity in Java.

To learn JDBC architecture.

To perform Insert, Update, Delete, and Select operations.

To use PreparedStatement for secure SQL execution.

To understand transaction management using commit().

To interact with the database using Java programs.



---

3. Problem Statement Explanation

This program is designed to store and manage student information in a database.

The application connects Java with a MySQL database named college. A table called student stores student records such as:

Roll Number

Student Name


The program provides a menu-driven system where the user can:

1. Add new student details


2. Update existing student information


3. Delete student records


4. View all student records


5. Exit the program



Prepared statements are used to safely execute SQL queries and avoid SQL injection problems. Transaction management is used to save changes permanently in the database.
