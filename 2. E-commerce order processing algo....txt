import java.util.Scanner;

class Ecommerce
{
    // Private data members
    private int productId;
    private String productName;
    private double price;
    private int quantity;

    // Default Constructor
    Ecommerce()
    {
        this.productId = 0;
        this.productName = "Not Available";
        this.price = 0;
        this.quantity = 0;
    }

    // Parameterized Constructor
    Ecommerce(int productId, String productName, double price, int quantity)
    {
        this.productId = productId;
        this.productName = productName;
        this.price = price;
        this.quantity = quantity;
    }

    // Public method to calculate amount
    public double calculateAmount()
    {
        return price * quantity;
    }

    // Public method to display product details
    public void displayProduct()
    {
        System.out.println(productId + "\t\t" +
                           productName + "\t\t" +
                           price + "\t" +
                           quantity + "\t" +
                           calculateAmount());
    }

    public static void main(String args[])
    {
        Scanner sc = new Scanner(System.in);

        // Products initialized using constructors
        Ecommerce p1 = new Ecommerce(101, "Laptop", 50000, 1);
        Ecommerce p2 = new Ecommerce(102, "Mouse", 500, 2);

        // User input product
        System.out.println("Enter User Product Details");

        System.out.print("Enter Product ID: ");
        int id = sc.nextInt();

        sc.nextLine();

        System.out.print("Enter Product Name: ");
        String name = sc.nextLine();

        System.out.print("Enter Product Price: ");
        double price = sc.nextDouble();

        System.out.print("Enter Quantity: ");
        int quantity = sc.nextInt();

        Ecommerce p3 = new Ecommerce(id, name, price, quantity);

        // Dynamic total calculation
        double total = p1.calculateAmount() +
                       p2.calculateAmount() +
                       p3.calculateAmount();

        double discount = 0;

        // Discount policy
        if(total >= 50000)
        {
            discount = total * 0.20;
        }
        else if(total >= 20000)
        {
            discount = total * 0.10;
        }
        else if(total >= 10000)
        {
            discount = total * 0.05;
        }

        double finalAmount = total - discount;

        // Invoice generation
        System.out.println("\n========= E-COMMERCE INVOICE =========");

        System.out.println("ID\t\tName\t\tPrice\tQty\tAmount");

        p1.displayProduct();
        p2.displayProduct();
        p3.displayProduct();

        System.out.println("--------------------------------------");
        System.out.println("Total Amount : " + total);
        System.out.println("Discount     : " + discount);
        System.out.println("Final Amount : " + finalAmount);

        sc.close();
    }
}
