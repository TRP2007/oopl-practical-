import java.util.Scanner;

class Book
{
    int bookId;
    String bookName;
    boolean isIssued;

    // Static variable
    static int totalBooks = 0;

    // Constructor
    Book(int id, String name)
    {
        bookId = id;
        bookName = name;
        isIssued = false;

        totalBooks++;
    }

    // Display book details
    void displayBook()
    {
        System.out.println("Book ID : " + bookId);
        System.out.println("Book Name : " + bookName);

        if(isIssued)
        {
            System.out.println("Status : Issued");
        }
        else
        {
            System.out.println("Status : Available");
        }
    }

    // Issue book
    void issueBook()
    {
        if(!isIssued)
        {
            isIssued = true;
            System.out.println(bookName + " issued successfully.");
        }
        else
        {
            System.out.println(bookName + " is already issued.");
        }
    }

    // Return book
    void returnBook()
    {
        if(isIssued)
        {
            isIssued = false;
            System.out.println(bookName + " returned successfully.");
        }
        else
        {
            System.out.println(bookName + " was not issued.");
        }
    }

    // Static method
    static void showTotalBooks()
    {
        System.out.println("Total Books : " + totalBooks);
    }
}

class LibraryManagement
{
    public static void main(String args[])
    {
        Scanner sc = new Scanner(System.in);

        // Books added directly in code
        Book b1 = new Book(101, "Java");
        Book b2 = new Book(102, "Python");
        Book b3 = new Book(103, "Data Structure");

        int choice;

        do
        {
            System.out.println("\n===== LIBRARY MANAGEMENT SYSTEM =====");
            System.out.println("1. View Books");
            System.out.println("2. Issue Book");
            System.out.println("3. Return Book");
            System.out.println("4. Total Books");
            System.out.println("5. Exit");

            System.out.print("Enter choice : ");
            choice = sc.nextInt();

            switch(choice)
            {
                case 1:

                    b1.displayBook();
                    System.out.println();

                    b2.displayBook();
                    System.out.println();

                    b3.displayBook();

                    break;

                case 2:

                    System.out.println("Enter Book ID to issue:");
                    int issueId = sc.nextInt();

                    if(issueId == 101)
                    {
                        b1.issueBook();
                    }
                    else if(issueId == 102)
                    {
                        b2.issueBook();
                    }
                    else if(issueId == 103)
                    {
                        b3.issueBook();
                    }
                    else
                    {
                        System.out.println("Book not found.");
                    }

                    break;

                case 3:

                    System.out.println("Enter Book ID to return:");
                    int returnId = sc.nextInt();

                    if(returnId == 101)
                    {
                        b1.returnBook();
                    }
                    else if(returnId == 102)
                    {
                        b2.returnBook();
                    }
                    else if(returnId == 103)
                    {
                        b3.returnBook();
                    }
                    else
                    {
                        System.out.println("Book not found.");
                    }

                    break;

                case 4:

                    Book.showTotalBooks();

                    break;

                case 5:

                    System.out.println("Program Ended.");
                    break;

                default:

                    System.out.println("Invalid Choice.");
            }

        } while(choice != 5);

        sc.close();
    }
}

# 🟦 **DEFINITION OF ALL CONCEPTS USED**

---

## 🔹 Class

A class is a blueprint used to define properties and behaviors of objects.

Example:

```java id="7y5s9d"
class Book
```

---

## 🔹 Object

An object is an instance of a class used to access data and methods.

Example:

```java id="k9v2qp"
Book b1 = new Book(101, "Java");
```

---

## 🔹 Constructor

A special method used to initialize objects automatically.

Example:

```java id="b7j2qa"
Book(int id, String name)
```

👉 Used when object is created.

---

## 🔹 Static Variable

A variable shared by all objects of a class.

Example:

```java id="p3n8xv"
static int totalBooks = 0;
```

👉 Same value for all objects.

---

## 🔹 Static Method

A method that belongs to class, not object.

Example:

```java id="x8m2lq"
static void showTotalBooks()
```

---

## 🔹 Method

A block of code that performs a specific task.

Example:

* displayBook()
* issueBook()
* returnBook()

---

## 🔹 Scanner Class

Used to take input from user.

---

## 🔹 Boolean

A datatype that stores true/false values.

Example:

```java id="q2w9ld"
boolean isIssued;
```

---

## 🔹 String

Used to store text data.

---

## 🔹 Control Statements

Used for decision making:

* if-else
