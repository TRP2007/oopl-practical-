import java.sql.*;
import java.util.Scanner;

class EmployeeJDBC
{
    public static void main(String args[])
    {
        Scanner sc = new Scanner(System.in);

        try
        {
            // Load Driver
            Class.forName("com.mysql.cj.jdbc.Driver");

            // Connect Database
            Connection con = DriverManager.getConnection(
                    "jdbc:mysql://localhost:3306/company",
                    "root",
                    "avcoe");

            int choice;

            do
            {
                System.out.println("\n1.Add");
                System.out.println("2.Update");
                System.out.println("3.Delete");
                System.out.println("4.View");
                System.out.println("5.Exit");
                System.out.print("Enter Choice: ");

                choice = sc.nextInt();

                switch(choice)
                {
                    case 1:

                        System.out.print("Enter ID: ");
                        int id = sc.nextInt();
                        sc.nextLine();

                        System.out.print("Enter Name: ");
                        String name = sc.nextLine();

                        System.out.print("Enter Salary: ");
                        double salary = sc.nextDouble();

                        PreparedStatement ps1 = con.prepareStatement(
                                "insert into employee values(?,?,?)");

                        ps1.setInt(1, id);
                        ps1.setString(2, name);
                        ps1.setDouble(3, salary);

                        ps1.executeUpdate();

                        System.out.println("Employee Added");
                        break;

                    case 2:

                        System.out.print("Enter ID: ");
                        int uid = sc.nextInt();

                        System.out.print("Enter New Salary: ");
                        double newsal = sc.nextDouble();

                        PreparedStatement ps2 = con.prepareStatement(
                                "update employee set salary=? where id=?");

                        ps2.setDouble(1, newsal);
                        ps2.setInt(2, uid);

                        ps2.executeUpdate();

                        System.out.println("Employee Updated");
                        break;

                    case 3:

                        System.out.print("Enter ID: ");
                        int did = sc.nextInt();

                        PreparedStatement ps3 = con.prepareStatement(
                                "delete from employee where id=?");

                        ps3.setInt(1, did);

                        ps3.executeUpdate();

                        System.out.println("Employee Deleted");
                        break;

                    case 4:

                        PreparedStatement ps4 = con.prepareStatement(
                                "select * from employee");

                        ResultSet rs = ps4.executeQuery();

                        while(rs.next())
                        {
                            System.out.println(
                                    rs.getInt(1) + " " +
                                    rs.getString(2) + " " +
                                    rs.getDouble(3));
                        }

                        break;

                    case 5:

                        System.out.println("Program End");
                        break;

                    default:

                        System.out.println("Invalid Choice");
                }

            } while(choice != 5);

            con.close();
        }

        catch(Exception e)
        {
            System.out.println(e);
        }
    }
}

//Practical No. — JDBC CRUD Operations Using Java and MySQL

Title

Employee Management System using Java JDBC and MySQL Database


---

1. Aim

To develop a Java application that connects with MySQL database using JDBC and performs CRUD operations (Create, Read, Update, Delete) on employee records.


---

2. Objective

To understand Java Database Connectivity (JDBC).

To connect Java application with MySQL database.

To perform Insert, Update, Delete, and Select operations.

To use PreparedStatement for secure query execution.

To manage employee information using database.



---

3. Problem Statement Explanation

This program is developed to manage employee records using Java and MySQL database.

The application connects Java program with a database named company. It contains a table called employee which stores:

Employee ID

Employee Name

Employee Salary


The program provides a menu-driven interface where the user can:

1. Add employee details


2. Update employee salary


3. Delete employee record


4. View all employee records


5. Exit the program



PreparedStatement is used to securely execute SQL queries and prevent SQL Injection attacks.

The program helps in storing and managing employee data efficiently in database form.
